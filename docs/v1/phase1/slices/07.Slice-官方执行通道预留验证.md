#### 5.3.4 执行器工厂

```typescript
// src/adapters/twitter/TwitterExecutorFactory.ts

export class TwitterExecutorFactory {
  /**
   * 根据配置创建执行器实例
   */
  static create(config: TwitterExecutorConfig): ITwitterExecutor {
    switch (config.type) {
      case 'localbridge':
        return new LocalBridgeExecutor(
          config.id,
          config.name,
          config.localBridge!,
        );
      
      case 'twitter-mcp':
        return new TwitterMCPExecutor(
          config.id,
          config.name,
          config.twitterMCP!,
        );
      
      default:
        throw new Error(`Unknown executor type: ${config.type}`);
    }
  }
  
  /**
   * 获取当前激活的执行器
   */
  static async getActiveExecutor(): Promise<ITwitterExecutor> {
    const settings = await settingsService.getSettings();
    const activeConfig = settings.twitterExecutors.find(
      (e) => e.id === settings.defaultTwitterExecutorId && e.enabled,
    );
    
    if (!activeConfig) {
      throw new Error('No active Twitter executor configured');
    }
    
    return this.create(activeConfig);
  }
}
```

```typescript
export interface ReplyTask {
  id: string;

  executor: {

    /** 可选的执行器 ID 列表 */
    availableExecutorIds: string[];
  };
  
  status: 'pending' | 'executing' | 'completed' | 'failed';
  createdAt: Date;
  updatedAt: Date;
}
```

#### 5.4.2 执行器路由逻辑

```typescript
// src/services/executorRouter.ts

export class ExecutorRouter {
  /**
   * 为任务选择合适的执行器
   */
  async selectExecutor(task: ReplyTask): Promise<ITwitterExecutor> {
    const settings = await settingsService.getSettings();
    
    // 1. 优先使用任务指定的执行器
    if (task.executor.currentExecutorId) {
      const config = settings.twitterExecutors.find(
        (e) => e.id === task.executor.currentExecutorId && e.enabled,
      );
      if (config) {
        return TwitterExecutorFactory.create(config);
      }
    }
    
    // 2. 使用默认执行器
    return TwitterExecutorFactory.getActiveExecutor();
  }
}
```
