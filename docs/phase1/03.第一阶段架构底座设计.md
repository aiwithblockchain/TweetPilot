```typescript
// src/ai/IAIProvider.ts
export interface AIOptions {
  model?: string;
  temperature?: number;
  maxTokens?: number;
  stopSequences?: string[];
  [key: string]: any;  // 扩展参数
}

export interface IAIProvider {
  /**
   * 生成文本
   * @param prompt - 提示词
   * @param options - AI 参数
   * @returns 生成的文本
   */
  generateText(prompt: string, options?: AIOptions): Promise<string>;

  /**
   * 流式生成文本（预留）
   * @param prompt - 提示词
   * @param options - AI 参数
   * @returns 异步迭代器
   */
  generateTextStream(prompt: string, options?: AIOptions): AsyncIterator<string>;
}
```

### 3.3 第一阶段实现

```typescript
// src/ai/ClaudeProvider.ts
import Anthropic from '@anthropic-ai/sdk';
import type { IAIProvider, AIOptions } from './IAIProvider';

export class ClaudeProvider implements IAIProvider {
  private client: Anthropic;

  constructor(apiKey: string) {
    this.client = new Anthropic({ apiKey });
  }

  async generateText(prompt: string, options?: AIOptions): Promise<string> {
    const response = await this.client.messages.create({
      model: options?.model || 'claude-3-5-sonnet-20241022',
      max_tokens: options?.maxTokens || 4096,
      temperature: options?.temperature,
      stop_sequences: options?.stopSequences,
      messages: [{ role: 'user', content: prompt }],
    });

    return response.content[0].type === 'text' 
      ? response.content[0].text 
      : '';
  }

  async *generateTextStream(prompt: string, options?: AIOptions): AsyncIterator<string> {
    const stream = await this.client.messages.stream({
      model: options?.model || 'claude-3-5-sonnet-20241022',
      max_tokens: options?.maxTokens || 4096,
      temperature: options?.temperature,
      messages: [{ role: 'user', content: prompt }],
    });

    for await (const chunk of stream) {
      if (chunk.type === 'content_block_delta' && chunk.delta.type === 'text_delta') {
        yield chunk.delta.text;
      }
    }
  }
}
```

### 4.4 Reply Agent 使用方式

```typescript
// src/agents/ReplyAgent.ts
import type { IAIProvider } from '../ai/IAIProvider';
import type { IKnowledgeBase } from '../knowledge/IKnowledgeBase';

export class ReplyAgent {
  constructor(
    private aiProvider: IAIProvider,
    private knowledgeBase: IKnowledgeBase,  // ✅ 依赖接口
  ) {}

  async generateReply(comment: CommentInput, options?: ReplyOptions): Promise<string> {
    // 搜索知识库（第一阶段返回空数组）
    const knowledge = await this.knowledgeBase.search(comment.content, {
      limit: 5,
      type: 'semantic',
    });

    // 构建 prompt（已支持知识注入）
    const prompt = this.buildPrompt(comment, knowledge, options);
    
    return await this.aiProvider.generateText(prompt);
  }

  private buildPrompt(
    comment: CommentInput, 
    knowledge: KnowledgeItem[], 
    options?: ReplyOptions
  ): string {
    let prompt = `回复这条评论: ${comment.content}`;
    
    // ✅ 第一阶段 knowledge 为空数组，这段逻辑不执行
    // ✅ 第二阶段 knowledge 有内容，这段逻辑自动生效
    if (knowledge.length > 0) {
      prompt += `\n\n参考知识:\n${knowledge.map(k => k.content).join('\n')}`;
    }
    
    return prompt;
  }
}
```

## 6. 底座 4: Reply Agent 可扩展接口

### 6.1 架构依据

- **工作包 D 要求**: "Reply Agent 要支持不同角色生成不同风格回复"
- **工作包 D 要求**: "Reply Agent 要输出候选回复、风险建议、执行建议"

### 6.2 设计方案

```typescript
// src/agents/ReplyAgent.ts
export interface ReplyOptions {
  role?: string;           // 角色（第一阶段可以不传）
  temperature?: number;    // AI 温度参数
  maxTokens?: number;      // 最大 token 数
  [key: string]: any;      // 扩展参数
}

export interface ReplyResult {
  reply: string;           // 生成的回复
  riskLevel: 'low' | 'medium' | 'high';  // 风险等级
  confidence: number;      // 置信度 0-1
  metadata?: Record<string, unknown>;
}

export class ReplyAgent {
  constructor(
    private aiProvider: IAIProvider,
    private knowledgeBase: IKnowledgeBase,
  ) {}

  /**
   * 生成回复
   * @param comment - 评论输入
   * @param options - 可选参数（第一阶段可以不传）
   * @returns 回复结果
   */
  async generateReply(
    comment: CommentInput, 
    options?: ReplyOptions  // ✅ 可选参数，向后兼容
  ): Promise<ReplyResult> {
    const knowledge = await this.knowledgeBase.search(comment.content);
    const prompt = this.buildPrompt(comment, knowledge, options);
    
    const reply = await this.aiProvider.generateText(prompt, {
      temperature: options?.temperature,
      maxTokens: options?.maxTokens,
    });

    return {
      reply,
      riskLevel: this.assessRisk(reply),
      confidence: 0.8,
    };
  }

  private buildPrompt(
    comment: CommentInput, 
    knowledge: KnowledgeItem[], 
    options?: ReplyOptions
  ): string {
    let prompt = `回复这条评论: ${comment.content}`;
    
    // ✅ 第一阶段 options?.role 为 undefined，这段逻辑不执行
    // ✅ 第二阶段传入 role，这段逻辑自动生效
    if (options?.role) {
      prompt += `\n\n角色设定: ${options.role}`;
    }
    
    if (knowledge.length > 0) {
      prompt += `\n\n参考知识:\n${knowledge.map(k => k.content).join('\n')}`;
    }
    
    return prompt;
  }

  private assessRisk(reply: string): 'low' | 'medium' | 'high' {
    // 简单的风险评估逻辑
    if (reply.includes('抱歉') || reply.includes('无法')) {
      return 'low';
    }
    return 'medium';
  }
}
```
