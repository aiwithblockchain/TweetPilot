### 3.1 定义 CandidateReply 领域对象

创建 `src/domain/candidateReply.ts`：

```typescript
export type CandidateReplyId = string;

export interface CandidateReply {
  id: CandidateReplyId;
  
  // 归属关系
  commentInputId: string;          // 属于哪条评论输入
  accountId: string;               // 属于哪个账号
  roleId?: string;                 // 使用了哪个角色
  workspaceId: string;             // 归属客户空间
  
  // 回复内容
  content: string;                 // 候选回复文本
  
  // 生成元信息
  riskLevel: 'low' | 'medium' | 'high';  // 风险等级
  confidence: number;              // 置信度 0-1
  modelSource: string;             // 模型来源（如 'claude'）
  knowledgeHits: number;           // 知识命中数
  
  // 时间戳
  generatedAt: Date;               // 生成时间
  
  // 扩展字段
  metadata?: Record<string, unknown>;
}

export interface CreateCandidateReplyParams {
  commentInputId: string;
  accountId: string;
  roleId?: string;
  workspaceId: string;
  content: string;
  riskLevel: 'low' | 'medium' | 'high';

  modelSource: string;

  metadata?: Record<string, unknown>;
}
```

### 3.3 实现内存版仓储

创建 `src/data/repositories/InMemoryCandidateReplyRepository.ts`：

```typescript
export class InMemoryCandidateReplyRepository implements ICandidateReplyRepository {
  private replies: Map<CandidateReplyId, CandidateReply> = new Map();
  
  async save(reply: CandidateReply): Promise<void> {
    this.replies.set(reply.id, reply);
  }
  
  async findById(id: CandidateReplyId): Promise<CandidateReply | null> {
    return this.replies.get(id) || null;
  }
  
  async findByCommentInput(commentInputId: string): Promise<CandidateReply[]> {
    return Array.from(this.replies.values())
      .filter(r => r.commentInputId === commentInputId)
      .sort((a, b) => b.generatedAt.getTime() - a.generatedAt.getTime());
  }
  
  // ... 实现其他方法
}
```

### 3.4 实现工厂函数

在 `src/domain/candidateReply.ts` 中添加：

```typescript
export function generateCandidateReplyId(): CandidateReplyId {
  // 使用 UUID v4 避免碰撞风险（浏览器原生支持，无需依赖）
  return crypto.randomUUID();
}

export function createCandidateReply(
  params: CreateCandidateReplyParams
): CandidateReply {
  return {
    id: generateCandidateReplyId(),
    commentInputId: params.commentInputId,
    accountId: params.accountId,
    roleId: params.roleId,
    workspaceId: params.workspaceId,
    content: params.content,
    riskLevel: params.riskLevel,
    confidence: params.confidence,
    modelSource: params.modelSource,
    knowledgeHits: params.knowledgeHits,
    generatedAt: new Date(),
    metadata: params.metadata,
  };
}
```
