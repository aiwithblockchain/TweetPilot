### 3.1 实现上下文组装逻辑

在 `ReplyAgent` 中实现 `assembleContext()` 方法：

```typescript
private async assembleContext(
  comment: CommentInput,
  options?: ReplyOptions
): Promise<GenerationContext> {
  // 1. 读取评论输入
  const commentContext = {
    content: comment.content,
    targetTweetId: comment.targetTweetId,
    targetTweetUrl: comment.targetTweetUrl,
  };
  
  // 2. 读取账号和角色
  const role = options?.role 
    ? await this.roleRepository.findById(options.role)
    : await this.roleRepository.getDefaultRole(comment.accountId);
  
  // 3. 读取最小历史互动（第一阶段可为空）
  const history = await this.getMinimalHistory(comment.accountId);
  
  // 4. 搜索知识库
  const knowledge = await this.knowledgeBase.search(comment.content, {
    limit: 5,
    type: 'semantic',
  });
  
  return {
    comment: commentContext,
    role,
    history,
    knowledge,
  };
}
```

### 3.2 实现 Prompt 构建逻辑

在 `ReplyAgent` 中实现 `buildPrompt()` 方法：

```typescript
private buildPrompt(context: GenerationContext): string {
  let prompt = `你需要回复以下评论:\n${context.comment.content}\n\n`;
  
  // 注入角色
  if (context.role) {
    prompt += `角色设定:\n${context.role.prompt}\n\n`;
  }
  
  // 注入知识
  if (context.knowledge.length > 0) {
    prompt += `参考知识:\n`;
    context.knowledge.forEach(k => {
      prompt += `- ${k.content}\n`;
    });
    prompt += `\n`;
  }
  
  // 注入历史（第一阶段可选）
  if (context.history && context.history.length > 0) {
    prompt += `历史互动:\n`;
    context.history.forEach(h => {
      prompt += `- ${h.summary}\n`;
    });
    prompt += `\n`;
  }
  
  prompt += `请生成一个合适的回复。`;
  
  return prompt;
}
```

### 3.3 实现候选回复生成逻辑

在 `ReplyAgent` 中实现 `generateReply()` 方法：

```typescript
async generateReply(
  comment: CommentInput,
  options?: ReplyOptions
): Promise<ReplyResult> {
  // 1. 组装上下文
  const context = await this.assembleContext(comment, options);
  
  // 2. 构建 prompt
  const prompt = this.buildPrompt(context);
  
  // 3. 调用 AI Provider 生成回复
  const reply = await this.aiProvider.generateText(prompt, {
    temperature: options?.temperature || 0.7,
    maxTokens: options?.maxTokens || 500,
  });
  
  // 4. 评估风险
  const riskLevel = this.assessRisk(reply);
  
  // 5. 计算置信度（第一阶段简单实现）
  const confidence = this.calculateConfidence(reply, context);
  
  return {
    reply,
    riskLevel,
    confidence,
    metadata: {
      modelSource: 'claude',
      knowledgeHits: context.knowledge.length,
      roleUsed: context.role?.id,
      generatedAt: new Date().toISOString(),
    },
  };
}
```

### 3.4 实现生成多个版本

添加 `generateMultipleReplies()` 方法：

```typescript
async generateMultipleReplies(
  comment: CommentInput,
  count: number = 3,
  options?: ReplyOptions
): Promise<ReplyResult[]> {
  const results: ReplyResult[] = [];
  
  for (let i = 0; i < count; i++) {
    // 使用不同的 temperature 生成不同版本
    const temperature = (options?.temperature || 0.7) + (i * 0.1);
    const result = await this.generateReply(comment, {
      ...options,
      temperature,
    });
    results.push(result);
  }
  
  return results;
}
```

### 3.5 实现简单风险评估

```typescript
private assessRisk(reply: string): 'low' | 'medium' | 'high' {
  // 简单的关键词匹配
  const highRiskKeywords = ['退款', '赔偿', '法律', '投诉'];
  const mediumRiskKeywords = ['抱歉', '问题', '错误'];
  
  const lowerReply = reply.toLowerCase();
  
  if (highRiskKeywords.some(k => lowerReply.includes(k))) {
    return 'high';
  }
  
  if (mediumRiskKeywords.some(k => lowerReply.includes(k))) {
    return 'medium';
  }
  
  return 'low';
}
```
